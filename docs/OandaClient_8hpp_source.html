219  template<class DataResponse, class HeartbeatResponse, typename ProcessData, typename ProcessHeartbeat>
220  void processBodyStream(
221  Concurrency::streams::istream& bodyStream,
222  ProcessData&& dataLambda,
223  ProcessHeartbeat&& heartbeatLambda,
224  pplx::cancellation_token_source& cts)
225  {
226  pplx::cancellation_token token = cts.get_token();
227  while (!bodyStream.is_eof() && !token.is_canceled()) {
228  Concurrency::streams::container_buffer<std::string> buffer;
230  bodyStream.read_to_delim(buffer, '\n')
231  .then([buffer, token](pplx::task<size_t> bufferSize) -> pplx::task<nlohmann::json> {
232  if (bufferSize.get() > 0) {
234  try {
235  return pplx::task_from_result(nlohmann::json::parse(buffer.collection()), token);
236  } catch(nlohmann::json::exception& e) {
237  #ifdef LOG_V20_ACCESS
238  LOG(ERROR) << "Parse Error: " << buffer.collection();
239  #endif
240  throw;
241  }
242  }
243  cancel_current_task();
244  return pplx::task_from_result(nlohmann::json(), token);
245  }).then([&, heartbeatLambda, dataLambda, token](pplx::task<nlohmann::json> parsedJson) -> pplx::task<void> {
246  auto json = parsedJson.get();
247  auto type = json.value("type", "not_recognized");
248  if (type.compare("HEARTBEAT") == 0) {
249  return createHeartbeatTask<HeartbeatResponse>(json, heartbeatLambda, token);
250  } else if (type.compare("not_recognized") != 0) {
251  return createDataTask<DataResponse>(json, dataLambda, token);
252  }
253  cancel_current_task();
254  return pplx::create_task([]{}, token);
255  }).wait();

418  void recurseAndFilterNullValues(nlohmann::json & jsonObject, std::string indent = "") {
419  if (!jsonObject.is_object() && !jsonObject.is_array()) {
420  //std::cout << indent << "Skipping object" << std::endl;
421  return;
422  }
423 
424  std::vector<nlohmann::json::object_t::key_type> keysToRemove;
425  for (auto &it : jsonObject.items()) {
426  //std::cout << indent << "Recurse object: " << it.key() << std::endl;
427  recurseAndFilterNullValues(it.value(), indent + "\t");
428  if (it.value().is_null() || (it.value().size() == 0)) {
429  //std::cout << indent << "Staging object: " << it.key() << std::endl;
430  keysToRemove.push_back(it.key());
431  }
432  }
433

435  //std::cout << indent << "Erasing object: " << key << std::endl;
436  jsonObject.erase(key);
